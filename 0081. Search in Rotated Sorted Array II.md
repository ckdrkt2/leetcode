```python
class Solution:
    def search(self, nums: List[int], target: int) -> bool:
        u, d = len(nums) - 1, 0
        while u >= d:
            m = (u + d) // 2
            if nums[m] == target: return True
            if nums[m] < nums[u]:
                if nums[m] < target <= nums[u]:
                    d = m + 1
                else:
                    u = m - 1
            elif nums[m] > nums[u]:
                if nums[d] <= target < nums[m]:
                    u = m - 1
                else:
                    d = m + 1
            else:
                u -= 1
        return False
```

time complexity: O(logN)      
space complexity: O(1)
