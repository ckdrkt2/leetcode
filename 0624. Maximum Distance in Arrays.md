```python
from typing import List

class Solution(object):
    def maxDistance(self, arrays: List[List[int]]) -> int:
        ans, m, M = 0, arrays[0][0], arrays[0][-1]

        for array in arrays[1:]:
            ans = max(ans, abs(M - array[0]), abs(m - array[-1]))
            m = min(m, array[0])
            M = max(M, array[-1])

        return ans
```
