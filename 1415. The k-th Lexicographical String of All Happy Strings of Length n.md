class Solution:
    def getHappyString(self, n: int, k: int) -> str:
        if k > 3 * 2 ** (n-1):
            return ""
        
        z = ("a", "b", "c")
        a = ""
        last = None
        i = 2 ** (n-1)
        
        while i > 0:
            b = (k-1) // i
            k = (k-1) % i
            if last:
                if last == z[b]: b += 1
                a += z[b]
                last = z[b]
            else:
                a += z[b]
                last = z[b]
            
            i = i // 2
        return a
