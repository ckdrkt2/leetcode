class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        a = sorted(boxTypes, key=lambda x : x[1], reverse=True); c = 0
        for i in a:
            if i[0] <= truckSize:
                truckSize -= i[0]
                c += i[0] * i[1]
            else:
                c += truckSize * i[1]
                break
        return c
