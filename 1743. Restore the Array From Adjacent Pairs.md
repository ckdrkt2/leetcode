```python
class Solution:
    def restoreArray(self, adjacentPairs: List[List[int]]) -> List[int]:
        graph, a = defaultdict(list), []
        for i, j in adjacentPairs:
            graph[i].append(j)
            graph[j].append(i)
        
        for i, j in graph.items():
            if len(j) == 1: break
        a += [i, j[0]]
        
        for i in range(len(adjacentPairs)-1):
            x, y = graph[a[-1]]
            if x == a[-2]: a.append(y)
            else: a.append(x)
        return a
```

time complexity: O(N)
space complexity: O(N)
