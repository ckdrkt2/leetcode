```python
from string import ascii_lowercase

class Solution:
    def getLucky(self, s: str, k: int) -> int:
        d = {a: str(i + 1) for i, a in enumerate(ascii_lowercase)}
        conv = ''.join(d[c] for c in s)
        for _ in range(k):
            conv = str(sum(int(i) for i in conv))
        return int(conv)
```
